//================================================================================
// Building script
//================================================================================
building_script city {
	city_hall
	shuffle start
		arena
		merchant
		food_seller
		blacksmith
		alchemist
		inn
		training_grounds
	shuffle end
	barracks
	set citizens = random(12,15)
	set citizens_world = citizens * 200 + random(0,199) + 100
	required off
	set count = citizens * 3
	group house
}

building_script village {
	village_hall
	set citizens = 5 + random(0,2)
	shuffle start
		merchant
		food_seller
		village_inn
		if rand 2
			blacksmith
			inc counter
			inc citizens
		endif
		if rand 2
			training_grounds
			inc counter
			inc citizens
		endif
		if counter < 2
			if rand 2
				alchemist
				inc citizens
			endif
		endif
	shuffle end
	barracks
	set citizens_world = citizens * 100 + random(0,99)
	required off
	set count = citizens
	group cottage
}
