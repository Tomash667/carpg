// merchant
stock merchant {
	// random items
	script {
		int price_limit;
		if(Level::IsCity())
			price_limit = Random(2000, 2500); // city
		else if(Level::IsVillage())
			price_limit = Random(500, 1000); // village
		else if(Level::IsTutorial())
			price_limit = 500; // tutorial
		else
			price_limit = 1000; // traveling merchant
		for(int i=0,count=Random(15,20); i<count; ++i)
		{
			ITEM_TYPE type = ITEM_TYPE(Rand() % 6);
			if(type == IT_CONSUMABLE)
				StockScript::AddRandomItem(type, price_limit/5, ITEM_NOT_SHOP | ITEM_NOT_MERCHANT, Random(2,5));
			else
				StockScript::AddRandomItem(type, price_limit, ITEM_NOT_SHOP | ITEM_NOT_MERCHANT);
		}
	}
	// potions
	random 5 10 p_nreg
	random 5 10 p_hp
}

// food seller - food & drinks and sometimes cooking items
stock food_seller {
	script {
		ItemList@ list = ItemList::Get("food_and_drink");
		bool is_village = !Level::IsCity();
		for(int i=0, count=list.Size(); i<count; ++i)
		{
			int value = Random(50, 100);
			if(is_village)
				value /= 2;
			Item@ item = list.GetItem(i);
			int add = value / item.value;
			if(add > 0)
				StockScript::AddItem(item, add);
		}
		if(Rand() % 4 == 0)
			StockScript::AddItem(Item::Get("frying_pan"));
		if(Rand() % 4 == 0)
			StockScript::AddItem(Item::Get("ladle"));
	}
}
